179    def apply_OD(self):
180        """
181        Extract OD from filename and calculate with its value what is the intensity before the OD filter
182        :return: self
183        """
184        match = re.search(r'_OD_([+-]?([0-9]*[.])?[0-9]+)__', self.FILEPATH)
185        if match==None:
186            self.OD = 0
187        else : 
188            self.OD = float(match.group(1))
189
190        self.scale_intensity(10**(self.OD))
191        return self

234    def fit_gauss(self, lambda0_guess, x_min = None, x_max = None, delta_lambda=.5, sigma_guess = .01, bounds = None, background_type = 'linear', plot=False):
235        """
236        Fits all Spectra of the Serie with a gaussian function (for all Spectra within [x_min, x_max])
237
238        :param lambda0_guess: (float) [nm] guess value of the position of the Voigt function
239        :param x_min: (float) [nm] minimum x values of the spectrum to be fitted
240        :param x_max: (float) [nm] maximum x values of the spectrum to be fitted
241        :param delta_lambda: (float) [nm] range of wavelengths centered around lambda0_guess, used to fit the Voigt profile
242        :param sigma_guess: (float) [nm] guess value of the standard deviation of the gaussian function
243        :param bounds: (dict) [nm] dictionary containing the limit values that the function parameters can reach during the fitting procedure. Available keys 
244            - "sigma_min"
245            - "sigma_max"
246        :param background_type: (str) defines the type of background function to fit. Accepted values are 
247            - "none"    : gaussian function only
248            - "const"   : gaussian function with constant 
249            - "linear"  : gaussian function with linear function (y = slope * lambda + intercept)
250        :param plot: (bool)  if True, plots the spectra and the fitted functions
251        :return: Dictionary of fit result or None if the fit failed
252        """
253
254        if x_min == None:
255            x_min = - numpy.infty
256        if x_max == None:
257            x_max = numpy.infty
258        first_fit = True
259        for i in range(len(self.spectra)): # starting with bigger x-values (high powers)
260            if (self.spectra[i].x > x_min) and (self.spectra[i].x < x_max):
261                if first_fit: # if first fit within this function, use the guess values provided when calling this function
262                    fit_result = self.spectra[i].fit_gauss(
263                            lambda0_guess           = lambda0_guess, 
264                            delta_lambda            = delta_lambda, 
265                            sigma_guess             = sigma_guess,
266                            background_type       = background_type,
267                            bounds                  = bounds,
268                            plot                    = plot
269                    )
270                else: # use results from last fit in the same Serie as guess values
271                    fit_result = self.spectra[i].fit_gauss(
272                            lambda0_guess           = self.fits[-1]['lambda0'], 
273                            delta_lambda           = delta_lambda, 
274                            sigma_guess            = self.fits[-1]['sigma'],
275                            background_type       = background_type,
276                            bounds                  = bounds,
277                            plot                    = plot
278                    )
279
280                if fit_result==None:
281                    print('Serie.fit_gauss() : i = '+str(i) + 'could not be fitted')
282                else: 
283                    self.fits.append(fit_result)
284                    first_fit = False
